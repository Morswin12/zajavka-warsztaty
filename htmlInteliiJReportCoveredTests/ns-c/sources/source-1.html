         this.salary = salary;
     }

         int i = this.name.compareTo(o.name);
         if (i == 0) {
             i += o.surName.compareTo(this.surName);
         }
         if (i == 0) {
             i += this.age - o.age;
         }
         if (i == 0) {
             double i1 = this.salary - o.salary;
             int temporary;
             if (i1 == 0.00) {
                 temporary = 0;

                 temporary = 1;
             } else {
                 temporary = -1;
             }
             i += temporary;
         }
         return i;
     }
