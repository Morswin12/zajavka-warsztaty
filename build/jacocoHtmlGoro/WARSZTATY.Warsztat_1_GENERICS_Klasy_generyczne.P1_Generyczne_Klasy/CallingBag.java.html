    static class Car {
        String name = "Car";
        private String color;
        private int year;

        public Car(String color, int year) {
            this.color = color;
            this.year = year;
        }

        public String getName() {
            return name;
        }

        public String getColor() {
            return color;
        }

        public int getYear() {

        }

        @Override
        public String toString() {
